/*ΕΡΩΤΗΜΑ 1*/
SELECT ENAME, DEPTNO FROM EMP
WHERE DEPTNO=10
ORDER BY COMM;

/*ΕΡΩΤΗΜΑ 2*/
SELECT ENAME, JOB, SAL FROM EMP
ORDER BY JOB ASC, SAL DESC;

/*ΕΡΩΤΗΜΑ 3*/
SELECT FORMAT (AVG(SAL),0)"SALARY",JOB,COUNT(*) "SUM" FROM EMP
GROUP BY JOB
HAVING COUNT(*) >=1;

/*ΕΡΩΤΗΜΑ 6*/
SELECT DNAME AS DEPARTMENT, MANAGER.ENAME AS MANAGER, EMPLOYEE.ENAME AS EMPLOYEE FROM EMP EMPLOYEE LEFT JOIN EMP MANAGER ON EMPLOYEE.MGR = MANAGER.EMPNO, DEPT 
WHERE EMPLOYEE.DEPTNO=DEPT.DEPTNO 
ORDER BY DEPARTMENT, EMPLOYEE;

/*ΕΡΩΤΗΜΑ 7*/
SELECT ENAME, JOB, LOC FROM DEPT, EMP
WHERE DNAME="RESEARCH" AND EMP.DEPTNO=DEPT.DEPTNO;
